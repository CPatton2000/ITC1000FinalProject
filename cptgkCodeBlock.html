This code block takes 2 user inputs and outputs if the words are anagrams of each other

    
def areAnagram(str1, str2):
    # Get lengths of both strings
    n1 = len(str1)
    n2 = len(str2)

    if n1 != n2:
        return 0
 
    str1 = sorted(str1)
    str2 = sorted(str2)
 
    for i in range(0, n1):
        if str1[i] != str2[i]:
            return 0
 
    return 1
 
 
str1 = "test"
str2 = "ttew"
 
if areAnagram(str1, str2):
    print("The two strings are anagram of each other")
else:
    print("The two strings are not anagram of each other")
